\begin{equation} \label{eq:energy}
E=\int_{0}^{1}\frac{1}{2}(\alpha|\mathbf{x}'(s)|^{2}+\beta|\mathbf{x}''(s)|^{2})+E_{ext}(\mathbf{x}(s)) \ud s
\end{equation}
